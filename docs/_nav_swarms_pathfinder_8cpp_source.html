13TArray<FVector> FNavSwarmsPathfinder::FindPath(FPathParameters& PathParameters, const bool& StopTask)
14{
15 PathParameters.PathVoxels.Empty(NavVoxels.Num());
16 for (int i = 0; i < NavVoxels.Num(); i++)
17 {
18 PathParameters.PathVoxels.Add(FPathVoxel());
19 }
20 const int StartIndex = NavDataAsset->GetIndexFromWorldVector(PathParameters.StartPosition);
21 const int EndIndex = NavDataAsset->GetIndexFromWorldVector(PathParameters.EndPosition);
22
23 FNavVoxel* StartVoxel = &NavVoxels[StartIndex];
24 if (StartVoxel->bIsObstacle)
25 {
26 float Distance = 10000000000000000;
27 for (const auto VoxelIndex : StartVoxel->Neighbours)
28 {
29 FNavVoxel& Voxel = NavVoxels[VoxelIndex];
30 float DistanceToNeighbour = FVector::Distance(PathParameters.StartPosition, Voxel.WorldLocation);
31 if (!Voxel.bIsObstacle && DistanceToNeighbour < Distance)
32 {
33 Distance = DistanceToNeighbour;
34 StartVoxel = &Voxel;
35 }
36 }
37 }
38 FNavVoxel* EndVoxel = &NavVoxels[EndIndex];
39
40 if (StartVoxel == nullptr || EndVoxel == nullptr || StartVoxel == EndVoxel || StartVoxel->bIsObstacle)
41 {
42 return {StartVoxel->WorldLocation, EndVoxel->WorldLocation};
43 }
44 FNavVoxel* CurrentVoxel = nullptr;
45
46 FBinaryHeap<float> Heap;
47 Heap.Add(0, StartVoxel->Index);
48
49 int Iterations = 0;
50 while (!Heap.IsEmpty())
51 {
52 const int TopIndex = Heap.Top();
53 CurrentVoxel = &NavVoxels[TopIndex];
54 FPathVoxel* CurrentPathVoxel = GetPathVoxelFromNavVoxel(PathParameters, *CurrentVoxel);
55
56 Iterations++;
57 if (StopTask || Iterations > 500000) break;
58
59 // We found a path, no need to search further
60 if (CurrentVoxel->Index == EndVoxel->Index) break;
61
62 CurrentPathVoxel->OnClosedList = true;
63 Heap.Pop();
64
65 const int TotalCost = CurrentPathVoxel->G_Cost + 8;
66 for (const auto VoxelIndex : CurrentVoxel->Neighbours)
67 {
68 FNavVoxel* NeighbourVoxel = &NavVoxels[VoxelIndex];
69 FPathVoxel* NeighbourPathVoxel = &PathParameters.PathVoxels[VoxelIndex];
70 // If the neighbour is an obstacle is occupied or is already in the closed set, skip the neighbour.
71 if (NeighbourVoxel->bIsObstacle || NeighbourPathVoxel->OnClosedList) continue;
72
73 // If the open set doesn't contain the current neighbour, add it
74 if (!Heap.IsPresent(NeighbourVoxel->Index))
75 {
76 NeighbourPathVoxel->ParentIndex = CurrentVoxel->Index;
77 NeighbourPathVoxel->H_Cost = FVector::Distance(EndVoxel->WorldLocation, NeighbourVoxel->WorldLocation);
78 // Heuristic Cost
79
80 Heap.Add(NeighbourPathVoxel->GetCost(), NeighbourVoxel->Index);
81 }
82 // Otherwise update the cost and set the parent as the current voxel
83 else if (TotalCost < NeighbourPathVoxel->G_Cost)
84 {
85 NeighbourPathVoxel->ParentIndex = CurrentVoxel->Index;
86 NeighbourPathVoxel->G_Cost = TotalCost;
87
88 Heap.Update(NeighbourPathVoxel->GetCost(), NeighbourVoxel->Index);
89 }
90 }
91 }
92 return GetPathData(CurrentVoxel, StartVoxel, PathParameters);
93}

95TArray<FVector> FNavSwarmsPathfinder::GetPathData(FNavVoxel* CurrentVoxel, FNavVoxel* StartVoxel,
96 FPathParameters& PathParameters)
97{
98 TArray<FVector> Path;
99 FVector PreviousLocation = CurrentVoxel->WorldLocation;
100
101 while (true)
102 {
103 const FVector CurrentLocation = CurrentVoxel->WorldLocation;
104
105 FNavVoxel* NextVoxel = GetNextVoxel(CurrentVoxel, PathParameters);
106 if (PathParameters.SmoothPath)
107 {
108 float Dot = 0;
109 if (NextVoxel != nullptr && PreviousLocation != CurrentVoxel->WorldLocation)
110 {
111 FVector DirectionToPrev = (CurrentLocation - PreviousLocation).GetSafeNormal();
112 FVector DirectionToNext = (NextVoxel->WorldLocation - CurrentLocation).GetSafeNormal();
113
114 Dot = FVector::DotProduct(DirectionToPrev, DirectionToNext);
115 }
116 if (Dot < PathParameters.SmoothAmount)
117 {
118 Path.Add(CurrentLocation);
119 PreviousLocation = CurrentLocation;
120 }
121 }
122 else
123 {
124 Path.Add(CurrentLocation);
125 }
126
127 CurrentVoxel = NextVoxel;
128
129 if (CurrentVoxel == nullptr)
130 {
131 break;
132 }
133 }
134
135 Path.Add(StartVoxel->WorldLocation);
136 Algo::Reverse(Path);
137 return Path;
138}
