5void UNavSwarmsDataAsset::BakeNavData(FIntVector GridSize, uint32 VoxelSize, FVector Location, const int32 Channels,
6 const TArray<AActor*>& IgnoredActors, const UWorld* World)
7{
8 TArray<FNavVoxel> NavVoxels;
9
10 const FVector GridBounds = FVector(GridSize.X, GridSize.Y, GridSize.Z) * VoxelSize * 0.5f;
11 const FVector VoxelBounds = FVector(VoxelSize) * 0.5f;
12 const FVector Center = Location + VoxelBounds - GridBounds;
13
14 InternalGridSize = GridSize;
15 InternalVoxelSize = VoxelSize;
16 InternalCenter = Center;
17
18 FCollisionShape CollisionShape = FCollisionShape::MakeBox(VoxelBounds);
19
20 FCollisionQueryParams CollisionQueryParams = FCollisionQueryParams(FName("NavCollisionQuery"));
21 CollisionQueryParams.AddIgnoredActors(IgnoredActors);
22 FCollisionObjectQueryParams ObjectQueryParams = FCollisionObjectQueryParams(Channels);
23
24 int Index = 0;
25 for (int y = 0; y < GridSize.Y; y++)
26 {
27 for (int x = 0; x < GridSize.X; x++)
28 {
29 for (int z = 0; z < GridSize.Z; z++)
30 {
31 const FIntVector SelfPosition(x, y, z);
32 FVector WorldPosition = Center + FVector(x, y, z) * VoxelSize;
33
34 TArray<FOverlapResult> OutOverlaps;
35
36 bool const bHit = World->OverlapMultiByObjectType(OutOverlaps, WorldPosition, FQuat::Identity,
37 ObjectQueryParams, CollisionShape,
38 CollisionQueryParams);
39
40 TArray<int> Neighbours = GetNeighbours(GridSize, SelfPosition);
41
42 FNavVoxel Voxel = FNavVoxel(WorldPosition, Index, false, bHit, Neighbours);
43 Index++;
44
45 NavVoxels.Add(Voxel);
46 }
47 }
48 }
49
50 InternalDepth = Index - 1;
51
52 FNavSwarmsSerializer::SaveToFile(NavVoxels, Name);
53 AttachedFilePath = FNavSwarmsSerializer::GetFilePath(Name);
54
55 if (MarkPackageDirty())
56 {
57 UE_LOG(LogTemp, Log, TEXT("Baked %d voxels"), NavVoxels.Num());
58 }
59}

94TArray<int> UNavSwarmsDataAsset::GetNeighbours(const FIntVector GridSize, const FIntVector SelfPosition) const
95{
96 TArray<int> Neighbours;
97 for (int y = -1; y <= 1; y++)
98 {
99 for (int x = -1; x <= 1; x++)
100 {
101 for (int z = -1; z <= 1; z++)
102 {
103 // Ignore self
104 if (x == 0 && y == 0 && z == 0) continue;
105
106 // if ((FMath::Abs(x) == 1 && (y != 0 || z != 0)) ||
107 // (FMath::Abs(y) == 1 && (x != 0 || z != 0)) ||
108 // (FMath::Abs(z) == 1 && (x != 0 || y != 0))) continue;
109
110 //if (FMath::Abs(x) == 1 && FMath::Abs(y) == 1 && FMath::Abs(z) == 1) continue;
111
112 int NX = SelfPosition.X + x;
113 int NY = SelfPosition.Y + y;
114 int NZ = SelfPosition.Z + z;
115
116 bool XInBounds = NX >= 0 && NX < GridSize.X;
117 bool YInBounds = NY >= 0 && NY < GridSize.Y;
118 bool ZInBounds = NZ >= 0 && NZ < GridSize.Z;
119
120 if (XInBounds && YInBounds && ZInBounds)
121 {
122 const FVector Pos = FVector(NX, NY, NZ);
123 Neighbours.Add(GetIndexFromLocalVector(Pos));
124 }
125 }
126 }
127 }
128
129 return Neighbours;
130}

137int UNavSwarmsDataAsset::GetIndexFromWorldVector(const FVector& InVec) const
138{
139 FVector LocalVector = (InVec - InternalCenter) / InternalVoxelSize;
140
141 LocalVector.X = FMath::RoundToInt(LocalVector.X);
142 LocalVector.Y = FMath::RoundToInt(LocalVector.Y);
143 LocalVector.Z = FMath::RoundToInt(LocalVector.Z);
144
145 if (const int Index = GetIndexFromLocalVector(LocalVector);
146 Index >= 0 && Index < InternalDepth)
147 {
148 return Index;
149 }
150 return 0;
