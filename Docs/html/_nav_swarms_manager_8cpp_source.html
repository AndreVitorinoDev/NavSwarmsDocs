57void ANavSwarmsManager::StartPathSearchAsync(const int32 OwnerID, const FVector Start, const FVector End,
58 const bool bIsPersistent, const float DelayBetweenSearches)
59{
60 if (!NavSwarmsDataAsset || NavVoxels.IsEmpty())
61 {
62 UE_LOG(NavSwarms_Log, Error, TEXT("NavSwarmsDataAsset not set or the grid is empty!"));
63 return;
64 }
65
66 // Check if the Actor has already a pathfinding task active, if it does returns
67 const bool OwnerRegistered = TaskDataMap.Contains(OwnerID);
68 if (OwnerRegistered)
69 {
70 if (!TaskDataMap.Find(OwnerID)->Runnable->IsFinished()) return;
71 }
72
73 const FPathParameters PathParameters = FPathParameters(Start, End, PathResolution, SmoothPath, SmoothAmount);
74 // Create new thread with the parameters (Copy, since we don't want to accidentally or not modify this)
75 FNavSwarmsRunnable* Runnable = new FNavSwarmsRunnable(PathParameters, this, OwnerID, bIsPersistent,
76 DelayBetweenSearches);
77
78 if (OwnerRegistered)
79 {
80 // Delete the thread or we'll be left with dangling pointers
81 TaskDataMap.Find(OwnerID)->Runnable->Kill();
82 TaskDataMap.Find(OwnerID)->Runnable = Runnable;
83 }
84 else
85 {
86 TaskDataMap.Add(OwnerID, FNavTaskData(OwnerID, Runnable));
87 }
88}

140FVector ANavSwarmsManager::GetPathPointAtDistance(const int32 OwnerID, const FVector ActorLocation, const float Distance)
141{
142 TArray<FVector>& Path = GetPathPoints(OwnerID);
143 if (Path.IsEmpty()) return FVector::ZeroVector;
144
145 int StartIndex = 0;
146 for (int32 i = 0; i < Path.Num() - 1; i++)
147 {
148 FVector DirectionToPoint = ActorLocation - Path[i];
149 DirectionToPoint.Normalize();
150
151 FVector DirectionOfPath = Path[i + 1] - Path[i];
152 DirectionOfPath.Normalize();
153
154 // Actor to point is facing the current path direction, meaning the point is behind
155 if (FVector::DotProduct(DirectionToPoint, DirectionOfPath) < 0)
156 {
157 StartIndex = i;
158 break;
159 }
160 }
161
162 float TotalDistance = FVector::Dist(ActorLocation, Path[StartIndex]);
163 for (int32 i = StartIndex; i < Path.Num(); i++)
164 {
165 if (i + 1 >= Path.Num())
166 return Path[i];
167
168 const float DistanceToNextPoint = FVector::Distance(Path[i], Path[i + 1]);
169
170 if (TotalDistance + DistanceToNextPoint >= Distance)
171 {
172 FVector DirectionOfPath = Path[i + 1] - Path[i];
173 DirectionOfPath.Normalize();
174
175 const float RemainingDistance = Distance - TotalDistance;
176
177 DrawDebugSphere(GetWorld(), Path[i] + (DirectionOfPath * RemainingDistance), 5, 12, FColor::Green);
178 return Path[i] + (DirectionOfPath * RemainingDistance);
179 }
180 TotalDistance += DistanceToNextPoint;
181 }
182
183 return Path[StartIndex];
184}

248void ANavSwarmsManager::BakeNavData()
249{
250 // Make sure there's a NavDataAsset to save information to
251 if (!NavSwarmsDataAsset)
252 {
253 UE_LOG(NavSwarms_Log, Error, TEXT("NavDataAsset not set!"));
254 return;
255 }
256 if (NavSwarmsDataAsset->Name.IsEmpty())
257 {
258 UE_LOG(NavSwarms_Log, Error, TEXT("File name not set on the DataAsset!"));
259 return;
260 }
261
262 // Stop all threads and clear everything
263 for (const auto Tasks : TaskDataMap)
264 {
265 RemoveFromNavManager(Tasks.Key, false);
266 }
267 TaskDataMap.Empty();
268
269 const FVector Location = GetActorLocation();
270
271 // Mix all the channels into a single bitmask
272 int32 Channels = 0;
273 for (auto CollisionChannel : ObstacleQueryChannels)
274 {
275 Channels |= ECC_TO_BITFIELD(CollisionChannel);
276 }
277
278 NavSwarmsDataAsset->BakeNavData(VoxelCount, VoxelSize, Location, Channels, ActorsToIgnoreForCollision, GetWorld());
279
280 // We could get back the data from the previous method...
281 NavVoxels = FNavSwarmsSerializer::LoadFromFile(NavSwarmsDataAsset->Name);
282 NavVoxelsOccupied.Empty();
283 for (const auto Voxel : NavVoxels)
284 {
285 NavVoxelsOccupied.Add(Voxel);
286 }
287
288 // This shouldn't be necessary
289 if (Pathfinder)
290 delete Pathfinder;
291 Pathfinder = new FNavSwarmsPathfinder(NavSwarmsDataAsset.Get(), NavVoxels);
292}
