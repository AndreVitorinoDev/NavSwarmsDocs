40{
41 for (int i = 0; i < NavVoxels.Num(); i++)
42 {
43 const FNavVoxel& Voxel = NavVoxels[i];
44 if (!Voxel.bIsObstacle || FVector::Distance(CameraPosition, Voxel.WorldLocation) > 2000) continue;
45 int NAround = 0;
46 for (int x = 0; x < Voxel.Neighbours.Num(); x++)
47 {
48 const FNavVoxel& NeighbourVoxel = NavVoxels[Voxel.Neighbours[x]];
49 if (NeighbourVoxel.bIsObstacle)
50 {
51 const bool SameX = NeighbourVoxel.WorldLocation.X == Voxel.WorldLocation.X;
52 const bool SameY = NeighbourVoxel.WorldLocation.Y == Voxel.WorldLocation.Y;
53 const bool SameZ = NeighbourVoxel.WorldLocation.Z == Voxel.WorldLocation.Z;
54
55 if ((SameX && SameY) || (SameX && SameZ) || (SameY && SameZ))
56 {
57 NAround++;
58 }
59 }
60 }
61 if (NAround >= 6) continue;

64 (
65 World,
66 Voxel.WorldLocation,
67 FVector(VoxelSize) * 0.5f,
68 FQuat::Identity,
69 FColor::Red,
70 false,
71 -1.f,
72 0,
